fs-single-extra-data-screen(WS: class kb-workspace, W: class g2-window)

{ Display the DCS input data screen - It has already been posted, but may be hidden, so we just need to display it again! }

DIALOG: class phgxl-spreadsheet-dialog;
BLOCK, OTHER-BLOCK: class s88-recipe-entity;
P: class ph-procedure-or-method = this procedure;
DEBUG-MODE: truth-value = the debug-mode of P;
BASE: truth-value;

begin
	if DEBUG-MODE then post "In [the name of P]";
 
	DIALOG = the phgxl-spreadsheet-dialog superior to WS;
	BLOCK = the s88-recipe-entity that is the-item-for-a-phgxl-dialog DIALOG;
	WS = the workspace of BLOCK;
	BASE = call s88-get (BLOCK, "base-update-allowed.val", s88-superior-operation);
	
	if DEBUG-MODE then post "In [the name of P] with BASE=[base].";

	if BASE = true then OTHER-BLOCK = fs-single-base-more-review else OTHER-BLOCK = fs-single-mod-more-review;

        if there exists a phgxl-spreadsheet-dialog DIALOG that is the-phgxl-dialog-for-an-item OTHER-BLOCK then begin
		if DEBUG-MODE then post "In [the name of P] Posting More Review Screen.";
		call ils-show-workspace (the subworkspace of DIALOG, W);
		end
	else begin
		if DEBUG-MODE then post "In [the name of P] with no dialog to post.";
	end;

end
